## 排序算法
### 基本数据结构
```python
class record():
    def __init__(self, key, datum):
        self.key = key
        self.datum = datum
```
对于排序而言，只关心record的key，默认为升序排序

### 插入排序（O(n^2)，稳定，适应）
```python
def insert_sort(lst):
    for i in range(1, len(lst)):
        x = lst[i]
        j = i
        while j > 0 and x.key < lst[j-1].key:
            lst[j] = lst[j-1]
            j -= 1
        lst[j] = x
```

### 选择排序（O(n^2)，不稳定，不适应）
```python
def select_sort(lst):
    for i in range(len(lst)-1):
        k = i # k记录最小下标
        for j in range(i,len(lst)):
            if lst[j].key < lst[k].key:
                k = j
        if i != k:
            lst[i], lst[k] = lst[k], lst[i]
```

### 堆排序（O(nlogn)，不稳定，不适应）
```python
def heap_sort(lst):
    def siftdown(lst, e, begin, end):
        i, j = begin, begin*2+1
        while j < end:
            if j + 1 < end and lst[j+1] < lst[j]:
                j += 1
            if e < lst[j]:
                break
            lst[i] = lst[j]
            i, j = j, j*2+1
        lst[i] = e
    
    end = len(lst)
    for i in range(end//2, -1, -1):  # build a heap
        siftdown(lst, lst[i], i, end)
    for i in range(end-1, -1, -1): # 将优先队列里的元素一个个放在后面
        e = lst[i]
        lst[i] = lst[0]
        siftdown(lst, e, 0, i)
```

def bubble_sort(lst): # 改进了适应
    for i in range(len(lst)):
        found = False
        for j in range(1, len(lst)-i):
            if lst[j].key < lst[j-1].key:
                lst[j], lst[j-1] = lst[j-1], lst[j]

### 快速排序（O(nlogn)，不稳定，不适应）
```python
def qsort_rec(lst, l, r):
    if l >= r:
        return
    i = l
    j = r
    pivot = lst[l]
    while i < j:
        while i < j and lst[j].key >= pivot.key:
            j -= 1
        if i < j:
            lst[i] = lst[j]
            i += 1
        while i < j and lst[i].key <= pivot.key:
            i += 1
        if i < j:
            lst[j] = lst[i]
            j -= 1
    lst[i] = pivot
    qsort_rec(lst, l, i-1)
    qsort_rec(lst, i+1, r)

def quick_sort(lst):
    qsort_rec(lst, 0, len(lst)-1)
```

快速排序的另一种简单实现：
```python
def quick_sort1(lst):
    def qsort_rec1(lst, l, r):
        if l >= r:
            return
        i = l
        pivot = lst[l].key  # pivot的record已知存放在lst[l],就不用额外的存储变量
        for j in range(l+1, r+1):
            if lst[j].key < pivot:
                i += 1
                lst[i], lst[j] = lst[j], lst[i]
            else:
                continue
        lst[i], lst[l] = lst[l], lst[i]
        qsort_rec1(lst, l, i-1)
        qsort_rec1(lst, i+1, r)
    
    qsort_rec1(lst, 0, len(lst)-1)
```

### 归并排序（O(nlogn)，稳定，不适应）
```python
def merge(lfrom, lto, low, mid, high):
    i, j, k = low, mid, low
    while i < mid and j < high:
        if lfrom[i] <= lfrom[j]:
            lto[k] = lfrom[i]
            i += 1
        else:
            lto[k] = lfrom[j]
            j += 1
        k += 1
    while i < mid:
        lto[k] = lfrom[i]
        i += 1
        k += 1
    while j < high:
        lto[k] = lfrom[j]
        j += 1
        k += 1

def merge_pass(lfrom, lto, llen, slen):
    i = 0
    while i + 2*slen < llen:
        merge(lfrom, lto, i, i+slen, i+2*slen)
        i += 2*slen
    if i + slen < llen:
        merge(lfrom, lto, i, i+slen, llen)
    else:
        # lto[i:llen] = lfrom[i:llen]
        for j in range(i, llen):
            lto[j] = lfrom[j]

def merge_sort(lst):
    slen = 1
    llen = len(lst)
    templist = [None]*llen
    while slen < llen:
        merge_pass(lst, templist, llen, slen)
        slen *= 2
        merge_pass(templist, lst, llen, slen)   # 就算完成排序时完整的序列存在templist中，只要最后经过这一个merge_pass就可以复制回原位置
        slen *= 2
```
